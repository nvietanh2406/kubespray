  --name argocd-apps \
  --repo https://git.datx.com.vn/datx-k8s/k8s-intervpc.git \
  --revision master \
  --path helm-charts/argocd-apps \
  --values values.yaml \
  --dest-namespace argocd-apps \
  --dest-server https://kubernetes.default.svc \
  --self-heal \
  --auto-prune \
  --sync-policy auto \
  --upsert
